class Solution:
    def pivotIndex(self, nums: List[int]) -> int:
        length = len(nums)
        all_sum = sum(nums)
        left_sum = 0
        for idx in range(length):
            if left_sum == all_sum - left_sum - nums[idx]:
                return idx
            left_sum += nums[idx]
        return -1
